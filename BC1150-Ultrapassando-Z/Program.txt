using System;
using System.Globalization;

namespace beecrowd
{
    class iniciante
    {
        static void Main(string[] args)
        {
            int i, X, Z, soma;

            X = int.Parse(Console.ReadLine());
            Z = int.Parse(Console.ReadLine());

            soma = 0;

            while (Z <= X)
            {
                Z = int.Parse(Console.ReadLine());
            }

            for (i = 0; soma < Z; i++)
            {
                soma = soma + X + i;
            }

            Console.WriteLine(i);
        }
    }
}
